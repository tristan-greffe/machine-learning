"""

Ce script couvre :

- Filtrage conditionnel avec isnull / notnull
- Suppression des donn√©es manquantes (dropna)
- Remplacement des donn√©es manquantes (fillna)

"""

import pandas as pd
import numpy as np

# ======================================================
# 1. √Ä quoi ressemblent les valeurs manquantes
# ======================================================

"""
Pandas utilise plusieurs repr√©sentations pour les donn√©es manquantes :

- np.nan  ‚Üí donn√©es num√©riques (float)
- pd.NA   ‚Üí valeur manquante g√©n√©rique (nullable)
- pd.NaT  ‚Üí donn√©es temporelles (timestamp)

pd.NA a √©t√© introduit pour unifier la gestion des valeurs manquantes
quel que soit le type de donn√©es.
"""

print("=== Valeurs manquantes ===")
print("np.nan :", np.nan)
print("pd.NA  :", pd.NA)
print("pd.NaT :", pd.NaT)
print()

# ------------------------------------------------------
# Comparaisons √† √©viter
# ------------------------------------------------------

print("=== Comparaisons incorrectes ===")
print("np.nan == np.nan :", np.nan == np.nan)
print("np.nan in [np.nan] :", np.nan in [np.nan])
print("np.nan is np.nan :", np.nan is np.nan)
print("pd.NA == pd.NA :", pd.NA == pd.NA)
print()

"""
‚ö†Ô∏è Conclusion :
- Ne JAMAIS utiliser == pour tester une valeur manquante
- Toujours utiliser isna() / notna()
"""

# ======================================================
# 2. Cr√©ation et exploration d'un DataFrame
# ======================================================

df = pd.read_csv("./data/movie_scores.csv")

print("=== DataFrame initial ===")
print(df)
print()

print("=== Valeurs manquantes (isnull) ===")
print(df.isnull())
print()

print("=== Valeurs pr√©sentes (notnull) ===")
print(df.notnull())
print()

# ======================================================
# 3. Filtrage conditionnel avec donn√©es manquantes
# ======================================================

print("=== Acteurs avec pr√©nom connu ===")
print(df[df["first_name"].notnull()])
print()

print("=== Score avant film manquant MAIS sexe connu ===")
condition = (
    df["pre_movie_score"].isnull()
    & df["sex"].notnull()
)
print(df[condition])
print()

# ======================================================
# 4. Suppression des donn√©es manquantes (dropna)
# ======================================================

print("=== Suppression des lignes avec au moins un NaN ===")
print(df.dropna())
print()

print("=== Suppression des lignes enti√®rement NaN ===")
print(df.dropna(how="all"))
print()

print("=== Suppression avec seuil (au moins 1 valeur non nulle) ===")
print(df.dropna(thresh=1))
print()

print("=== Suppression des colonnes avec NaN ===")
print(df.dropna(axis=1))
print()

print("=== Suppression des colonnes avec moins de 4 valeurs non nulles ===")
print(df.dropna(thresh=4, axis=1))
print()

# ======================================================
# 5. Remplacement des donn√©es manquantes (fillna)
# ======================================================

print("=== Remplacement global (d√©conseill√©) ===")
print(df.fillna("NEW VALUE!"))
print()

print("=== Remplacement cibl√© : first_name ===")

print(df["first_name"])
print()

print("=== Remplacement par la moyenne ===")

print("Moyenne pre_movie_score :", mean_score)

print()

print("=== Remplacement automatique (num√©rique uniquement) ===")

# ======================================================
# 6. Interpolation
# ======================================================

"""
L‚Äôinterpolation permet d‚Äôestimer une valeur manquante
en fonction des valeurs voisines.
‚ö†Ô∏è √Ä utiliser uniquement si l‚Äôordre des donn√©es est pertinent.
