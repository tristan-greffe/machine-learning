Les DataFrames Pandas ‚Äã

Un DataFrame pandas est une structure de donn√©es tabulaire :

  • lignes ‚Üí observations

"""
dataframe_basics.py
==================
Bases des DataFrames Pandas
Ce script couvre :
- La cr√©ation de DataFrames √† partir de tableaux ou de fichiers CSV
- L'acc√®s aux colonnes et aux lignes
- La manipulation des donn√©es : cr√©ation, suppression, modification

import pandas as pd
import numpy as np

# ======================================================
# 1. Cr√©ation d'un DataFrame √† partir d'objets Python
# ======================================================

# Fixer la seed pour reproduire les m√™mes nombres al√©atoires
np.random.seed(101)

# Cr√©ation d'un tableau 4x3 avec des entiers al√©atoires entre 0 et 100
mydata = np.random.randint(0, 101, (4, 3))
print("Tableau de donn√©es al√©atoires :\n", mydata, "\n")

# D√©finir un index pour les lignes (exemple : abr√©viations des √©tats)
myindex = ['CA', 'NY', 'AZ', 'TX']

# D√©finir les noms des colonnes
mycolumns = ['Jan', 'Feb', 'Mar']

# Cr√©ation du DataFrame √† partir des donn√©es uniquement
df = pd.DataFrame(data=mydata)
print("DataFrame avec index et colonnes par d√©faut :\n", df, "\n")

# Cr√©ation du DataFrame avec un index personnalis√©
df = pd.DataFrame(data=mydata, index=myindex)
print("DataFrame avec index personnalis√© :\n", df, "\n")

# Cr√©ation du DataFrame avec index et colonnes personnalis√©s
df = pd.DataFrame(data=mydata, index=myindex, columns=mycolumns)
print("DataFrame complet avec index et colonnes :\n", df, "\n")

# Obtenir des informations sur le DataFrame
print("Informations g√©n√©rales sur le DataFrame :")
df.info()
print("\n")

# ======================================================
# 2. Lecture d'un fichier CSV dans un DataFrame
# ======================================================

# Lecture du fichier 'tips.csv' (assurez-vous qu'il est dans le m√™me dossier)
df = pd.read_csv('./data/tips.csv')

print("Colonnes :", df.columns)
print("Index :", df.index)
print("Premi√®res lignes :", df.head(3))
print("Derni√®res lignes :", df.tail(3))
print("R√©sum√© info :")
df.info()
print("Nombre de lignes :", len(df))
print("Statistiques descriptives :")
print(df.describe().transpose())
print("\n")

# Saisir plusieurs colonnes
print("Acc√®s √† plusieurs colonnes :")

df['tip_percentage'] = 100 * df['tip'] / df['total_bill']

# Arrondir les valeurs √† 2 d√©cimales
df['price_per_person'] = np.round(df['price_per_person'], 2)

print("DataFrame avec nouvelles colonnes :\n", df.head(), "\n")

# Supprimer une colonne
df = df.drop('tip_percentage', axis=1)
print("DataFrame apr√®s suppression de la colonne 'tip_percentage' :\n", df.head(), "\n")

# ======================================================
# 4. Bases de l'index
# ======================================================

# V√©rifier l'index actuel
print("Index actuel :", df.index, "\n")

# D√©finir une colonne comme index
df = df.set_index('Payment ID')
print("DataFrame apr√®s avoir d√©fini 'Payment ID' comme index :\n", df.head(), "\n")

# R√©initialiser l'index (transforme l'index en colonne)
df = df.reset_index()
print("DataFrame apr√®s r√©initialisation de l'index :\n", df.head(), "\n")

# ======================================================
# 5. Manipulation des lignes (rows)
# ======================================================

# D√©finir √† nouveau 'Payment ID' comme index pour manipuler les lignes
df = df.set_index('Payment ID')

# Acc√®s √† une seule ligne
print("Acc√®s √† la premi√®re ligne par position (iloc) :\n", df.iloc[0], "\n")
print("Acc√®s √† une ligne par label (loc) :\n", df.loc['Sun2959'], "\n")

# Acc√®s √† plusieurs lignes
print("Acc√®s √† plusieurs lignes par position :\n", df.iloc[0:4], "\n")
print("Acc√®s √† plusieurs lignes par label :\n", df.loc[['Sun2959','Sun5260']], "\n")

# Supprimer une ligne par label
df_temp = df.drop('Sun2959', axis=0)
print("DataFrame apr√®s suppression de la ligne 'Sun2959' (temporaire) :\n", df_temp.head(), "\n")

# Ins√©rer une nouvelle ligne (rarement utilis√© en pratique)
# Copier une ligne existante

# Attention : .append() est d√©pr√©ci√© depuis Pandas 2.0, utiliser pd.concat() √† la place

print("Lignes o√π total_bill > 40 :\n", df_filtered1.head(), "\n")

# Filtrage direct sans variable interm√©diaire
df_filtered_direct = df[df['total_bill'] > 40]
print("Filtrage direct total_bill > 40 :\n", df_filtered_direct.head(), "\n")

# Exemple : filtrage sur une colonne cat√©gorielle
df_filtered2 = df[df['sex'] == 'Male']

# ======================================================
print("=== Filtrage conditionnel : conditions multiples ===\n")
# Exemple : total_bill > 30 ET sex = 'Male'
df_filtered_and = df[(df['total_bill'] > 30) & (df['sex'] == 'Male')]
print("total_bill > 30 ET sex = 'Male' :\n", df_filtered_and.head(), "\n")

# Exemple : total_bill > 30 OU sex = 'Female'
df_filtered_or = df[(df['total_bill'] > 30) | (df['sex'] == 'Female')]
print("total_bill > 30 OU sex = 'Female' :\n", df_filtered_or.head(), "\n")

# ======================================================
print("=== Filtrage conditionnel : plusieurs valeurs avec isin ===\n")
# Exemple : filtrer les jours du week-end (Saturday, Sunday)
weekend_days = ['Saturday', 'Sunday']
df_filtered_weekend = df[df['day'].isin(weekend_days)]
print("Lignes correspondant au week-end :\n", df_filtered_weekend.head(), "\n")

# Ajouter un jour suppl√©mentaire (Friday)

# ======================================================
# 7. M√©thodes utiles : apply sur une seule colonne
# ======================================================

print("=== M√©thode apply : appliquer une fonction custom sur une colonne ===\n")

# Exemple 1 : extraire les 4 derniers chiffres d'une colonne 'cc_number' (num√©ro de carte)
# On cr√©e une fonction custom
def last_four(num):
    """Retourne les 4 derniers chiffres d'un nombre en tant que cha√Æne"""
    return str(num)[-4:]

# V√©rification rapide de la fonction
print("Test fonction last_four sur un nombre :", last_four(123456789), "\n")

# Application de la fonction √† la colonne 'cc_number' via apply
# Attention : cette colonne doit exister dans votre DataFrame df
if 'cc_number' in df.columns:
    df['last_four'] = df['cc_number'].apply(last_four)
    print("Exemple d'application de last_four sur la colonne 'cc_number' :\n", df[['cc_number', 'last_four']].head(), "\n")
else:
    print("Colonne 'cc_number' non pr√©sente dans df. Passez √† l'exemple suivant.\n")

# Exemple 2 : cat√©goriser le total_bill en $ (low, medium, high) avec apply
def categorize_price(price):
    """Retourne un label $ en fonction du prix"""
    if price < 10:
        return '$'
    elif 10 <= price < 30:
        return '$$'
    else:
        return '$$$'

# Application de la fonction sur la colonne 'total_bill'
df['price_category'] = df['total_bill'].apply(categorize_price)
print("DataFrame avec nouvelle colonne 'price_category' :\n", df[['total_bill', 'price_category']].head(), "\n")

# Note p√©dagogique :
print("""
üí° Points cl√©s sur l'utilisation de apply sur une seule colonne :
- La fonction pass√©e √† apply doit prendre en entr√©e **une seule valeur** (une ligne de la s√©rie)
- La fonction doit **retourner une seule valeur**, pas une s√©rie
- Tr√®s utile pour transformer ou cr√©er de nouvelles colonnes √† partir d'une colonne existante
""")

# ======================================================
# 8. M√©thodes utiles : apply sur des colonnes multiples
# ======================================================

print("=== M√©thode apply : appliquer une fonction sur plusieurs colonnes ===\n")

# ------------------------------------------------------
# Rappel : expression lambda
# ------------------------------------------------------
# lambda x: x * 2
# Fonction anonyme √† usage unique, tr√®s utilis√©e avec apply


# ------------------------------------------------------
# Exemple : √©valuer la qualit√© du pourboire
# Colonnes utilis√©es : total_bill et tip
# ------------------------------------------------------

def tip_quality(total_bill, tip):
    """
    √âvalue la qualit√© du pourboire en fonction du ratio tip / total_bill
    """
    if (tip / total_bill) > 0.25:
        return "G√©n√©reux"
    else:
        return "Standard"


# Test rapide de la fonction
print("Test tip_quality :", tip_quality(16.99, 1.01), "\n")


# ------------------------------------------------------
# M√©thode 1 : apply + lambda (axis=1)
# ------------------------------------------------------
df['tip_quality_apply'] = df[['total_bill', 'tip']].apply(
    lambda row: tip_quality(row['total_bill'], row['tip']),
    axis=1
)

print("R√©sultat avec apply + lambda :\n",
      df[['total_bill', 'tip', 'tip_quality_apply']].head(), "\n")


# ------------------------------------------------------
# M√©thode 2 : np.vectorize (plus lisible et souvent plus rapide)
# ------------------------------------------------------
import numpy as np

df['tip_quality_vectorized'] = np.vectorize(tip_quality)(
    df['total_bill'],
    df['tip']
)

print("R√©sultat avec np.vectorize :\n",
      df[['total_bill', 'tip', 'tip_quality_vectorized']].head(), "\n")


# ------------------------------------------------------
# Notes p√©dagogiques importantes
# ------------------------------------------------------

üí° Points cl√©s √† retenir :
- apply sur plusieurs colonnes n√©cessite axis=1
- La fonction custom doit retourner UNE seule valeur par ligne
- lambda est pratique pour des appels ponctuels
- np.vectorize rend une fonction Python "consciente" de NumPy
- np.vectorize am√©liore souvent la lisibilit√© et parfois les performances
""")

groupby ‚Äã

groupby() sert √† regrouper des lignes partageant une m√™me valeur (segmenter les donn√©es par cat√©gories), puis √† appliquer un calcul (mean, sum, count‚Ä¶) sur chaque groupe.

üëâ Exactement comme en SQL

Exemple

Colonnes cat√©gorielles vs continues :

  • Colonnes cat√©gorielles : Valeurs discr√®tes mais peuvent √™tre num√©riques ou textuelles (ex : ann√©es)
  • Colonnes continues : Valeurs num√©riques continues

INFO

py
df.groupby('model_year')

√Ä ce stade :

  • Aucun calcul n‚Äôest effectu√©
  • Pandas cr√©e un objet GroupBy en attente
  • Il faut obligatoirement appeler une m√©thode d‚Äôagr√©gation

Les fonctions d‚Äôagr√©gation ‚Äã

Une fois les donn√©es regroup√©es, Pandas doit savoir quoi calculer. üîπ Fonctions les plus courantes

M√©thodeDescription
mean()Moyenne
sum()Somme
count()Nombre de valeurs
size()Taille du groupe
min() / max()Valeurs extr√™mes
std()√âcart-type
var()Variance
describe()Statistiques compl√®tes

Comprendre le MultiIndex ‚Äã

Lorsque tu groupes sur plusieurs colonnes, Pandas cr√©e un index hi√©rarchique (MultiIndex).

Fichier Python associ√©
py
"""
dataframe_groupby.py
====================
Op√©rations groupby et index multi-niveaux (MultiIndex)

Ce script couvre :
- Le fonctionnement de groupby()
- Les agr√©gations classiques
- Le groupby sur plusieurs colonnes
- Les index hi√©rarchiques (MultiIndex)
- La s√©lection avec loc et xs
- Le tri et la manipulation des niveaux
- Les agr√©gations avanc√©es avec agg()
"""

import pandas as pd
import numpy as np

# ======================================================
# 0. Chargement des donn√©es
# ======================================================

df = pd.read_csv("./data/mpg.csv")

print("=== Aper√ßu du DataFrame ===")
print(df.head())
print()

# ======================================================
# 1. M√©thode groupby()
# ======================================================

"""
groupby() cr√©e un objet interm√©diaire.
Aucune op√©ration n‚Äôest effectu√©e tant qu‚Äôune m√©thode
d‚Äôagr√©gation n‚Äôest pas appliqu√©e.
"""

grouped = df.groupby("model_year")

print("Type de l'objet groupby :", type(grouped))
print()

# ------------------------------------------------------
# Agr√©gations classiques
# ------------------------------------------------------

print("=== Moyennes par ann√©e ===")
avg_year = grouped.mean(numeric_only=True)
print(avg_year)
print()

print("Index :", avg_year.index)
print("Colonnes :", avg_year.columns)
print()

print("=== Colonne mpg uniquement ===")
print(avg_year["mpg"])
print()

# Acc√®s direct en une ligne
print("=== mpg moyen par ann√©e (one-liner) ===")
print(df.groupby("model_year").mean(numeric_only=True)["mpg"])
print()

# Statistiques descriptives
print("=== describe() par ann√©e ===")
print(df.groupby("model_year").describe())
print()

print("=== describe() transpos√© ===")
print(df.groupby("model_year").describe().transpose())
print()

# ======================================================
# 2. Groupby sur plusieurs colonnes
# ======================================================

print("=== Moyennes par ann√©e ET par cylindres ===")
year_cyl = df.groupby(["model_year", "cylinders"]).mean(numeric_only=True)
print(year_cyl.head())
print()

print("Index du DataFrame multi-niveaux :")
print(year_cyl.index)
print()

# ======================================================
# 3. MultiIndex (index hi√©rarchique)
# ======================================================

print("=== Informations sur le MultiIndex ===")
print("Niveaux :", year_cyl.index.levels)
print("Noms des niveaux :", year_cyl.index.names)
print()

# ------------------------------------------------------
# S√©lection avec loc
# ------------------------------------------------------

print("=== Toutes les cylindr√©es pour l'ann√©e 70 ===")
print(year_cyl.loc[70])
print()

print("=== Ann√©es 70 et 72 ===")
print(year_cyl.loc[[70, 72]])
print()

print("=== Ligne unique : ann√©e 70, 8 cylindres ===")
print(year_cyl.loc[(70, 8)])
print()

# ------------------------------------------------------
# S√©lection transversale avec xs()
# ------------------------------------------------------

"""
xs() (cross-section) permet de s√©lectionner des donn√©es
√† partir d‚Äôun niveau interne du MultiIndex.
"""

print("=== Toutes les donn√©es pour l'ann√©e 70 (xs) ===")
print(year_cyl.xs(key=70, level="model_year"))
print()

print("=== Moyennes pour les moteurs 4 cylindres (toutes ann√©es) ===")
four_cyl = year_cyl.xs(key=4, level="cylinders")
print(four_cyl)
print()

"""
‚ö†Ô∏è Bon r√©flexe :
Il est souvent plus simple de FILTRER les donn√©es
avant le groupby plut√¥t que d‚Äôessayer de manipuler
le MultiIndex apr√®s coup.
"""

print("=== Filtrage AVANT groupby (6 et 8 cylindres) ===")
filtered = df[df["cylinders"].isin([6, 8])]
print(
    filtered
    .groupby(["model_year", "cylinders"])
    .mean(numeric_only=True)
)
print()

# ======================================================
# 4. Manipulation des niveaux du MultiIndex
# ======================================================

# √âchange des niveaux
print("=== Swap des niveaux ===")
print(year_cyl.swaplevel().head())
print()

# Tri du MultiIndex
print("=== Tri par ann√©e (descendant) ===")
print(year_cyl.sort_index(level="model_year", ascending=False).head())
print()

print("=== Tri par cylindres (descendant) ===")
print(year_cyl.sort_index(level="cylinders", ascending=False).head())
print()

# ======================================================
# 5. Agr√©gations avanc√©es avec agg()
# ======================================================

"""
agg() permet d‚Äôappliquer plusieurs fonctions
d‚Äôagr√©gation simultan√©ment, √©ventuellement diff√©rentes
selon les colonnes.
"""

# S√©lection des colonnes num√©riques
num_df = df.select_dtypes(include="number")

print("=== Agr√©gations globales ===")
print(num_df.agg(["median", "mean"]))
print()

print("=== Agr√©gations cibl√©es (mpg, weight) ===")
print(num_df.agg(["sum", "mean"])[["mpg", "weight"]])
print()

print("=== Agr√©gations sp√©cifiques par colonne ===")
print(
    df.agg({

Fusion ‚Äã

merge() fonctionne comme une jointure de base de donn√©es :

  • il utilise une cl√©,√† savoir how
  • il compare les valeurs
  • il d√©cide quelles lignes garder

Les types de jointures

  • INNER JOIN (intersection) : Garde uniquement les valeurs pr√©sentes dans les deux tables
  • LEFT JOIN : Garde toutes les lignes de gauche et compl√®te avec NaN si n√©cessaire
  • RIGHT JOIN : Sym√©trique du LEFT JOIN
  • OUTER JOIN (union compl√®te) : Garde tout ce qui existe au moins une fois

py
"""
dataframe_combination.py
========================
Combinaison de DataFrames avec Pandas

Ce script couvre :
- La concat√©nation avec pd.concat()
- Les fusions avec pd.merge()
- Les jointures SQL-like : inner, left, right, outer
- Les fusions sur index ou colonnes diff√©rentes
- La gestion des colonnes dupliqu√©es
"""

import pandas as pd
import numpy as np

# ======================================================
# 1. CONCAT√âNATION (pd.concat)
# ======================================================

"""
La concat√©nation permet de "coller" des DataFrames
ayant une structure similaire.
"""

data_one = {
    'A': ['A0', 'A1', 'A2', 'A3'],
    'B': ['B0', 'B1', 'B2', 'B3']
}

data_two = {
    'C': ['C0', 'C1', 'C2', 'C3'],
    'D': ['D0', 'D1', 'D2', 'D3']
}

one = pd.DataFrame(data_one)
two = pd.DataFrame(data_two)

print("=== DataFrame ONE ===")
print(one, "\n")

print("=== DataFrame TWO ===")
print(two, "\n")

# ------------------------------------------------------
# Concat√©nation par lignes (axis=0) ‚Äî par d√©faut
# ------------------------------------------------------

print("=== Concat√©nation par lignes (axis=0) ===")
concat_rows = pd.concat([one, two], axis=0)
print(concat_rows, "\n")

"""
‚ö†Ô∏è Les colonnes ne correspondent pas :
Pandas remplit automatiquement avec NaN
"""

# ------------------------------------------------------
# Concat√©nation par colonnes (axis=1)
# ------------------------------------------------------

print("=== Concat√©nation par colonnes (axis=1) ===")
concat_cols = pd.concat([one, two], axis=1)
print(concat_cols, "\n")

# ------------------------------------------------------
# Concat√©nation par lignes AVEC colonnes identiques
# ------------------------------------------------------

print("=== Concat√©nation par lignes avec colonnes align√©es ===")
two.columns = one.columns  # renommage C,D ‚Üí A,B
concat_aligned = pd.concat([one, two], axis=0)
print(concat_aligned, "\n")

# R√©initialisation de l'index
concat_aligned = concat_aligned.reset_index(drop=True)
print("=== Index r√©initialis√© ===")
print(concat_aligned, "\n")

# ======================================================
# 2. FUSION (pd.merge)
# ======================================================

"""
pd.merge() fonctionne comme une jointure SQL
"""

registrations = pd.DataFrame({
    'reg_id': [1, 2, 3, 4],
    'name': ['Andrew', 'Bobo', 'Claire', 'David']
})

logins = pd.DataFrame({
    'log_id': [1, 2, 3, 4],
    'name': ['Xavier', 'Andrew', 'Yolanda', 'Bobo']
})

print("=== Registrations ===")
print(registrations, "\n")

print("=== Logins ===")
print(logins, "\n")

# ------------------------------------------------------
# INNER JOIN
# ------------------------------------------------------

print("=== INNER JOIN ===")
inner_merge = pd.merge(
    registrations,
    logins,
    how='inner',
    on='name'
)
print(inner_merge, "\n")

"""
INNER :
‚Üí uniquement les cl√©s pr√©sentes dans LES DEUX tables
"""

# ------------------------------------------------------
# LEFT JOIN
# ------------------------------------------------------

print("=== LEFT JOIN ===")
left_merge = pd.merge(
    registrations,
    logins,
    how='left',
    on='name'
)
print(left_merge, "\n")

"""
LEFT :
‚Üí toutes les lignes de la table de gauche
"""

# ------------------------------------------------------
# RIGHT JOIN
# ------------------------------------------------------

print("=== RIGHT JOIN ===")
right_merge = pd.merge(
    registrations,
    logins,
    how='right',
    on='name'
)
print(right_merge, "\n")

"""
RIGHT :
‚Üí toutes les lignes de la table de droite
"""

# ------------------------------------------------------
# OUTER JOIN
# ------------------------------------------------------

print("=== OUTER JOIN ===")
outer_merge = pd.merge(
    registrations,
    logins,
    how='outer',
    on='name'
)
print(outer_merge, "\n")

"""
OUTER :
‚Üí union compl√®te des deux tables
"""

# ======================================================
# 3. FUSION SUR INDEX
# ======================================================

registrations_idx = registrations.set_index("name")

print("=== Fusion index (left_index=True) ===")
merge_index = pd.merge(
    registrations_idx,
    logins,
    left_index=True,
    right_on='name',
    how='inner'
)
print(merge_index, "\n")

# ======================================================
# 4. COLONNES CL√âS DIFF√âRENTES
# ======================================================

registrations_renamed = registrations.rename(columns={'name': 'reg_name'})

print("=== Colonnes diff√©rentes ===")
print(registrations_renamed, "\n")

merge_diff_cols = pd.merge(
    registrations_renamed,
    logins,
    left_on='reg_name',
    right_on='name',
    how='inner'
)

print(merge_diff_cols, "\n")

# Nettoyage
merge_diff_cols = merge_diff_cols.drop('reg_name', axis=1)
print("=== Apr√®s nettoyage ===")
print(merge_diff_cols, "\n")

# ======================================================
# 5. COLONNES DUPLIQU√âES & SUFFIXES
# ======================================================

registrations_dup = registrations.rename(columns={'reg_id': 'id'})
logins_dup = logins.rename(columns={'log_id': 'id'})

print("=== Colonnes dupliqu√©es ===")
print(pd.merge(registrations_dup, logins_dup, on='name'), "\n")

print("=== Suffixes personnalis√©s ===")
print(
    pd.merge(
        registrations_dup,
        logins_dup,
        on='name',
        suffixes=('_reg', '_log')
    )
)

"""
pandas_pivot_tables.py
=====================
Le pivotement des donn√©es peut aider √† clarifier les relations et √† explorer vos datasets. 

Documentation compl√®te sur le pivot et les m√©thodes connexes :
https://pandas.pydata.org/docs/user_guide/reshaping.html

import numpy as np
import pandas as pd

# ======================================================
# Lecture du fichier CSV
# ======================================================
df = pd.read_csv('./data/Sales_Funnel_CRM.csv')

# ======================================================
# 1. M√©thode pivot()
# ======================================================
"""
La m√©thode pivot() r√©organise les donn√©es en fonction des valeurs de colonnes et d'un nouvel index.
‚ö†Ô∏è Note : Il n'est pas toujours n√©cessaire de pivoter les donn√©es. Cette m√©thode est surtout utile
pour l'analyse, l'exploration et la visualisation.

Checklist avant d'utiliser pivot() :
1. Quelle question essayez-vous de r√©pondre ?
2. √Ä quoi ressemblerait le DataFrame qui r√©pond √† cette question ?
3. Le pivot est-il vraiment n√©cessaire ?

help(pd.pivot)  # Documentation rapide

# Exemple pratique : combien de licences de chaque produit Google a-t-il achet√© ?
licenses = df[['Company', 'Product', 'Licenses']]

# Pivot simple
pivot_simple = pd.pivot(
    data=licenses,
    index='Company',   # valeurs uniques de Company deviennent l'index
    columns='Product', # valeurs uniques de Product deviennent les colonnes
    values='Licenses'  # valeurs num√©riques √† afficher
)
pivot_simple

# ======================================================
# 2. M√©thode pivot_table()
# ======================================================
"""
Comme pivot(), pivot_table() r√©organise les donn√©es mais permet √©galement d'appliquer des fonctions

pivot_sum[['Licenses', 'Sale Price']]
